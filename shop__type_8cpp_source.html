58 bool ShopType::Load(RcdFileReader *rcd_file, const ImageMap &sprites, const TextMap &texts)
59 {
60  if (rcd_file->version != 4 || rcd_file->size != 2 + 1 + 1 + 4 * 4 + 3 * 4 + 4 * 4 + 2 + 4) return false;
61  uint16 width = rcd_file->GetUInt16();
62  this->height = rcd_file->GetUInt8();
63  if (this->height != 1) return false; // Other heights may fail.
64  this->flags = rcd_file->GetUInt8() & 0xF;
65 
66  for (int i = 0; i < 4; i++) {
67  ImageData *view;
68  if (!LoadSpriteFromFile(rcd_file, sprites, &view)) return false;
69  if (width != 64) continue; // Silently discard other sizes.
70  this->views[i] = view;
71  }
72 
73  for (uint i = 0; i < 3; i++) {
74  uint32 recolour = rcd_file->GetUInt32();
75  this->recolours.Set(i, RecolourEntry(recolour));
76  }
77  this->item_cost[0] = rcd_file->GetInt32();
78  this->item_cost[1] = rcd_file->GetInt32();
79  this->monthly_cost = rcd_file->GetInt32();
80  this->monthly_open_cost = rcd_file->GetInt32();
81 
82  uint8 val = rcd_file->GetUInt8();
83  if (!IsValidItemType(val)) return false;
84  this->item_type[0] = (ItemType)val;
85 
86  val = rcd_file->GetUInt8();
87  if (!IsValidItemType(val)) return false;
88  this->item_type[1] = (ItemType)val;
89 
90  TextData *text_data;
91  if (!LoadTextFromFile(rcd_file, texts, &text_data)) return false;
92  StringID base = _language.RegisterStrings(*text_data, _shops_strings_table);
93  this->SetupStrings(text_data, base, STR_GENERIC_SHOP_START, SHOPS_STRING_TABLE_END, SHOPS_NAME_TYPE, SHOPS_DESCRIPTION_TYPE);
94  return true;
95 }
