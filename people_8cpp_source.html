43 static bool IsGoodEdgeRoad(int16 x, int16 y)

55 static Point16 FindEdgeRoad()
56 {
57  int16 highest_x = _world.GetXSize() - 1;
58  int16 highest_y = _world.GetYSize() - 1;
59  for (int16 x = 1; x < highest_x; x++) {
60  if (IsGoodEdgeRoad(x, 0)) return {x, 0};
61  if (IsGoodEdgeRoad(x, highest_y)) return {x, highest_y};
62  }
63  for (int16 y = 1; y < highest_y; y++) {
64  if (IsGoodEdgeRoad(0, y)) return {0, y};
65  if (IsGoodEdgeRoad(highest_x, y)) return {highest_x, y};
66  }
67 
68  return {-1, -1};
69 }

174 void Guests::DoTick()
175 {
176  this->daily_frac++;
177  int end_index = std::min(this->daily_frac * GUEST_BLOCK_SIZE / TICK_COUNT_PER_DAY, GUEST_BLOCK_SIZE);
178  while (this->next_daily_index < end_index) {
179  Guest *p = this->block.Get(this->next_daily_index);
180  if (p->IsActive() && !p->DailyUpdate()) {
181  p->DeActivate(OAR_REMOVE);
182  this->AddFree(p);
183  }
184  this->next_daily_index++;
185  }
186  if (this->next_daily_index >= GUEST_BLOCK_SIZE) {
187  this->daily_frac = 0;
188  this->next_daily_index = 0;
189  }
190 }

197 void Guests::OnNewDay()
198 {
199  PersonType ptype = PERSON_PILLAR;
200  if (!this->CanUsePersonType(ptype)) return;
201  if (this->CountActiveGuests() >= _scenario.max_guests) return;
202  if (!this->rnd.Success1024(_scenario.GetSpawnProbability(512))) return;
203 
204  if (!IsGoodEdgeRoad(this->start_voxel.x, this->start_voxel.y)) {
205  /* New guest, but no road. */
206  this->start_voxel = FindEdgeRoad();
207  if (!IsGoodEdgeRoad(this->start_voxel.x, this->start_voxel.y)) return;
208  }
209 
210  if (!this->HasFreeGuests()) return; // No more quests available.
211  /* New guest! */
212  Guest *g = this->GetFree();
213  g->Activate(this->start_voxel, ptype);
214 }
