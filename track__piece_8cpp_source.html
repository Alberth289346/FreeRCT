121 static bool LoadTrackCurve(RcdFileReader *rcdfile, TrackCurve **curve, uint32 *length)
122 {
123 #define ENSURE_LENGTH(x) do { if (*length < (x)) { *curve = nullptr; return false; } *length -= (x); } while(false)
124 
125  ENSURE_LENGTH(1);
126  uint8 type = rcdfile->GetUInt8();
127  switch (type) {
128  case 0: // No track curve available.
129  *curve = nullptr;
130  return true;
131 
132  case 1: { // Curve consisting of a fixed value.
133  ENSURE_LENGTH(2);
134  int value = rcdfile->GetInt16();
135  *curve = new ConstantTrackCurve(value);
136  return true;
137  }
138 
139  case 2: {
140  ENSURE_LENGTH(1);
141  int count = rcdfile->GetUInt8();
142  ENSURE_LENGTH(count * 16u);
143  BezierTrackCurve *bezier = new BezierTrackCurve();
144  bezier->curve.reserve(count);
145  for (; count > 0; count--) {
146  bezier->curve.push_back(LoadBezier(rcdfile));
147  }
148  bezier->curve.shrink_to_fit();
149  *curve = bezier;
150  return true;
151  }
152 
153  default: // Error.
154  fprintf(stderr, "Unexpected curve type %u.\n", type);
155  *curve = nullptr;
156  return false;
157  }
158 #undef ENSURE_LENGTH
159 }

191 bool TrackPiece::Load(RcdFileReader *rcd_file, const ImageMap &sprites)
192 {
193  uint32 length = rcd_file->size;
194  if (rcd_file->version != 5 || length < 2 + 3 + 1 + 2 + 4 + 2) return false;
195  length -= 2 + 3 + 1 + 2 + 4 + 2;
196 
197  this->entry_connect = rcd_file->GetUInt8();
198  this->exit_connect = rcd_file->GetUInt8();
199  this->exit_dx = rcd_file->GetInt8();
200  this->exit_dy = rcd_file->GetInt8();
201  this->exit_dz = rcd_file->GetInt8();
202  this->speed = rcd_file->GetInt8();
203  this->track_flags = rcd_file->GetUInt16();
204  this->cost = rcd_file->GetUInt32();
205  this->voxel_count = rcd_file->GetUInt16();
206 
207  if (length < 36u * this->voxel_count) return false;
208  length -= 36u * this->voxel_count;
209 
210  this->track_voxels = new TrackVoxel[this->voxel_count];
211  for (int i = 0; i < this->voxel_count; i++) {
212  if (!this->track_voxels[i].Load(rcd_file, 36, sprites)) return false;
213  }
214  if (length < 4u) return false;
215  length -= 4;
216  this->piece_length = rcd_file->GetUInt32();
217 
218  bool ok = true;
219  ok = ok && LoadTrackCurve(rcd_file, &this->car_xpos, &length);
220  ok = ok && LoadTrackCurve(rcd_file, &this->car_ypos, &length);
221  ok = ok && LoadTrackCurve(rcd_file, &this->car_zpos, &length);
222  ok = ok && LoadTrackCurve(rcd_file, &this->car_pitch, &length);
223  ok = ok && LoadTrackCurve(rcd_file, &this->car_roll, &length);
224  ok = ok && LoadTrackCurve(rcd_file, &this->car_yaw, &length);
225  if (!ok || this->car_xpos == nullptr || this->car_ypos == nullptr || this->car_zpos == nullptr || this->car_roll == nullptr) return false;
226  return length == 0;
227 }

255 bool PositionedTrackPiece::IsOnWorld() const
256 {
257  assert(this->piece != nullptr);
258  if (!IsVoxelInsideWorld(this->x_base, this->y_base, this->z_base)) return false;
259  if (!IsVoxelInsideWorld(this->x_base + this->piece->exit_dx, this->y_base + this->piece->exit_dy,
260  this->z_base + this->piece->exit_dz)) return false;
261  const TrackVoxel *tvx = this->piece->track_voxels;
262  for (int i = 0; i < this->piece->voxel_count; i++) {
263  if (!IsVoxelInsideWorld(this->x_base + tvx->dx, this->y_base + tvx->dy, this->z_base + tvx->dz)) return false;
264  tvx++;
265  }
266  return true;
267 }

274 bool PositionedTrackPiece::CanBePlaced() const
275 {
276  if (!this->IsOnWorld()) return false;
277  const TrackVoxel *tvx = this->piece->track_voxels;
278  for (int i = 0; i < this->piece->voxel_count; i++) {
279  /* Is the voxel above ground level? */
280  if (_world.GetGroundHeight(this->x_base + tvx->dx, this->y_base + tvx->dy) > this->z_base + tvx->dz) return false;
281  const Voxel *vx = _world.GetVoxel(this->x_base + tvx->dx, this->y_base + tvx->dy, this->z_base + tvx->dz);
282  if (vx != nullptr && !vx->CanPlaceInstance()) return false;
283  }
284  return true;
285 }
