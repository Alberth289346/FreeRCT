51 bool CarType::Load(RcdFileReader *rcdfile, const ImageMap &sprites)
52 {
53  if (rcdfile->version != 2 || rcdfile->size != 2 + 2 + 4 + 4 + 2 + 2 + 16384) return false;
54 
55  this->tile_width = rcdfile->GetUInt16();
56  if (this->tile_width != 64) return false; // Do not allow anything else than 64 pixels tile width.
57 
58  this->z_height = rcdfile->GetUInt16();
59  if (this->z_height != this->tile_width / 4) return false;
60 
61  this->car_length = rcdfile->GetUInt32();
62  if (this->car_length > 65535) return false; // Assumption is that a car fits in a single tile, at least some of the time.
63 
64  this->inter_car_length = rcdfile->GetUInt32();
65  this->num_passengers = rcdfile->GetUInt16();
66  this->num_entrances = rcdfile->GetUInt16();
67  if (this->num_entrances == 0 || this->num_entrances > 4) return false; // Seems like a nice arbitrary upper limit on the number of rows of a car.
68  uint16 pass_per_row = this->num_passengers / this->num_entrances;
69  if (this->num_passengers != pass_per_row * this->num_entrances) return false;
70 
71  for (uint i = 0; i < 4096; i++) {
72  if (!LoadSpriteFromFile(rcdfile, sprites, &this->cars[i])) return false;
73  }
74  return true;
75 }

95 bool CoasterType::Load(RcdFileReader *rcd_file, const TextMap &texts, const TrackPiecesMap &piece_map)
96 {
97  uint32 length = rcd_file->size;
98  if (rcd_file->version != 5 || length < 2 + 1 + 1 + 1 + 4 + 2) return false;
99  length -= 2 + 1 + 1 + 1 + 4 + 2;
100  this->coaster_kind = rcd_file->GetUInt16();
101  this->platform_type = rcd_file->GetUInt8();
102  this->max_number_trains = rcd_file->GetUInt8();
103  this->max_number_cars = rcd_file->GetUInt8();
104  if (this->coaster_kind == 0 || this->coaster_kind >= CST_COUNT) return false;
105  if (this->platform_type == 0 || this->platform_type >= CPT_COUNT) return false;
106 
107  TextData *text_data;
108  if (!LoadTextFromFile(rcd_file, texts, &text_data)) return false;
109  StringID base = _language.RegisterStrings(*text_data, _coasters_strings_table);
110  this->SetupStrings(text_data, base, STR_GENERIC_COASTER_START, COASTERS_STRING_TABLE_END, COASTERS_NAME_TYPE, COASTERS_DESCRIPTION_TYPE);
111 
112  int piece_count = rcd_file->GetUInt16();
113  if (length != 4u * piece_count) return false;
114 
115  this->pieces.resize(piece_count);
116  for (auto &piece : this->pieces) {
117  uint32 val = rcd_file->GetUInt32();
118  if (val == 0) return false; // We don't expect missing track pieces (they should not be included at all).
119  auto iter = piece_map.find(val);
120  if (iter == piece_map.end()) return false;
121  piece = (*iter).second;
122  }
123  /* Setup a track voxel list for fast access in the type. */
124  this->voxel_count = 0;
125  for (const auto &piece : this->pieces) {
126  this->voxel_count += piece->voxel_count;
127  }
128  this->voxels = new const TrackVoxel *[this->voxel_count]();
129  int vi = 0;
130  for (const auto &piece : this->pieces) {
131  for (int j = 0; j < piece->voxel_count; j++) {
132  this->voxels[vi] = &piece->track_voxels[j];
133  vi++;
134  }
135  }
136  return true;
137 }

251 void DisplayCoasterCar::Set(int16 xvoxel, int16 yvoxel, int8 zvoxel, int16 xpos, int16 ypos, int16 zpos, uint8 pitch, uint8 roll, uint8 yaw)
252 {
253  bool change_voxel = this->x_vox != xvoxel || this->y_vox != yvoxel || this->z_vox != zvoxel;
254 
255  if (!change_voxel && this->x_pos == xpos && this->y_pos == ypos && this->z_pos == zpos &&
256  this->pitch == pitch && this->roll == roll && this->yaw == yaw) return; // Nothing changed.
257 
258  if (this->yaw != 0xff && change_voxel) {
259  /* Valid data, and changing voxel -> remove self from the old voxel. */
260  this->MarkDirty();
261  Voxel *v = _world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false);
262  this->RemoveSelf(v);
263  }
264 
265  /* Update voxel and orientation. */
266  this->x_vox = xvoxel;
267  this->y_vox = yvoxel;
268  this->z_vox = zvoxel;
269 
270  this->x_pos = xpos;
271  this->y_pos = ypos;
272  this->z_pos = zpos;
273 
274  this->pitch = pitch;
275  this->roll = roll;
276  this->yaw = yaw;
277 
278  if (this->yaw != 0xff) {
279  this->MarkDirty(); // Voxel or orientation has changed, repaint the possibly new voxel.
280 
281  if (change_voxel) { // With a really new voxel, also add self to the new voxel.
282  Voxel *v = _world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false);
283  this->AddSelf(v);
284  }
285  }

339 static void inline Unroll(uint roll, int32 *dy, int32 *dz)
340 {
341  /* Rotates around X axis, thus dx does not change. */
342  int32 new_dy = *dy * _sin_cos_table[roll + 4] - *dz * _sin_cos_table[roll];
343  int32 new_dz = *dy * _sin_cos_table[roll] + *dz * _sin_cos_table[roll + 4];
344  *dy = new_dy;
345  *dz = new_dz;
346 }

352 void CoasterTrain::OnAnimate(int delay)
353 {
354  if (this->speed >= 0) {
355  this->back_position += this->speed * delay;
356  if (this->back_position >= this->coaster->coaster_length) {
357  this->back_position -= this->coaster->coaster_length;
358  this->cur_piece = this->coaster->pieces;
359  }
360  while (this->cur_piece->distance_base + this->cur_piece->piece->piece_length < this->back_position) this->cur_piece++;
361  } else {
362  uint32 change = -this->speed * delay;
363  if (change > this->back_position) {
364  this->back_position = this->back_position + this->coaster->coaster_length - change;
365  /* Update the track piece as well. There is no simple way to get
366  * the last piece, so moving from the front will have to do. */
367  this->cur_piece = this->coaster->pieces;
368  while (this->cur_piece->distance_base + this->cur_piece->piece->piece_length < this->back_position) this->cur_piece++;
369  } else {
370  this->back_position -= change;
371  while (this->cur_piece->distance_base > this->back_position) this->cur_piece--;
372  }
373  }
374  uint32 car_length = this->coaster->car_type->car_length;
375  uint32 position = this->back_position; // Back position of the train / last car.
376  const PositionedTrackPiece *ptp = this->cur_piece;
377  for (uint i = 0; i < this->cars.size(); i++) {
378  CoasterCar &car = this->cars[i];
379  if (position >= this->coaster->coaster_length) {
380  position -= this->coaster->coaster_length;
381  ptp = this->coaster->pieces;
382  }
383  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
384 
385  /* Get position of the back of the car. */
386  int32 xpos_back = ptp->piece->car_xpos->GetValue(position - ptp->distance_base) + (ptp->x_base << 8);
387  int32 ypos_back = ptp->piece->car_ypos->GetValue(position - ptp->distance_base) + (ptp->y_base << 8);
388  int32 zpos_back = ptp->piece->car_zpos->GetValue(position - ptp->distance_base) * 2 + (ptp->z_base << 8);
389 
390  /* Get roll from the center of the car. */
391  position += car_length / 2;
392  if (position >= this->coaster->coaster_length) {
393  position -= this->coaster->coaster_length;
394  ptp = this->coaster->pieces;
395  }
396  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
397  uint roll = static_cast<uint>(ptp->piece->car_roll->GetValue(position - ptp->distance_base) + 0.5) & 0xf;
398 
399  /* Get position of the front of the car. */
400  position += car_length / 2;
401  if (position >= this->coaster->coaster_length) {
402  position -= this->coaster->coaster_length;
403  ptp = this->coaster->pieces;
404  }
405  while (ptp->distance_base + ptp->piece->piece_length < position) ptp++;
406  int32 xpos_front = ptp->piece->car_xpos->GetValue(position - ptp->distance_base) + (ptp->x_base << 8);
407  int32 ypos_front = ptp->piece->car_ypos->GetValue(position - ptp->distance_base) + (ptp->y_base << 8);
408  int32 zpos_front = ptp->piece->car_zpos->GetValue(position - ptp->distance_base) * 2 + (ptp->z_base << 8);
409 
410  int32 xder = xpos_front - xpos_back;
411  int32 yder = ypos_front - ypos_back;
412  int32 zder = (zpos_front - zpos_back) / 2; // Tile height is half the width.
413 
414  int32 xpos_middle = xpos_back + xder / 2; // Compute center point of the car as position to render the car.
415  int32 ypos_middle = ypos_back + yder / 2;
416  int32 zpos_middle = zpos_back + zder;
417 
418  float total_speed = sqrt(xder * xder + yder * yder + zder * zder);
419  /* Gravity */
420  this->speed -= zder / total_speed * 9.8;
421 
424  if (this->speed < 65536 / 1000 && (this->cur_piece->piece->HasPower() || this->cur_piece->piece->HasPlatform())) {
425  this->speed = 65536 / 1000;
426  }
427 
428  /* Unroll the orientation vector. */
429  Unroll(roll, &yder, &zder);
430  float horizontal_speed = std::hypot(xder, yder);
431 
432  static const double TAN11_25 = 0.198912367379658; // tan(11.25 degrees).
433  static const double TAN33_75 = 0.6681786379192989; // tan(3*11.25 degrees).
434 
435  /* Compute pitch. */
436  bool swap_dz = false;
437  if (zder < 0) {
438  swap_dz = true;
439  zder = -zder;
440  }
441 
442  uint pitch;
443  if (horizontal_speed < zder) {
444  if (horizontal_speed < zder * TAN11_25) {
445  pitch = 4;
446  } else if (horizontal_speed < zder * TAN33_75) {
447  pitch = 3;
448  } else {
449  pitch = 2;
450  }
451  } else {
452  if (zder < horizontal_speed * TAN11_25) {
453  pitch = 0;
454  } else if (zder < horizontal_speed * TAN33_75) {
455  pitch = 1;
456  } else {
457  pitch = 2;
458  }
459  }
460  if (swap_dz) pitch = (16 - pitch) & 0xf;
461 
462  /* Compute yaw. */
463  bool swap_dx = false;
464  if (xder > 0) {
465  swap_dx = true;
466  xder = -xder;
467  }
468  bool swap_dy = false;
469  if (yder > 0) {
470  swap_dy = true;
471  yder = -yder;
472  }
473  uint yaw;
474  /* There are 16 yaw directions, the 360 degrees needs to be split in 16 pieces.
475  * However the x and y axes are in the middle of a piece, so 360 degrees is split
476  * in 32 parts, and 2 parts form one piece. */
477  if (xder < yder) {
478  /* In the first 45 degrees. It is split in 4 parts (4*11.25 degrees)
479  * where the 1st part is for direction 0. The 2nd and 3rd part are for direction 1,
480  * and the 4th part is for direction 2. */
481  if (xder * TAN11_25 < yder) {
482  yaw = 0;
483  } else if (xder * TAN33_75 < yder) {
484  yaw = 1;
485  } else {
486  yaw = 2;
487  }
488  } else {
489  /* In the second 45 degrees. It is also split in 4 parts (4*11.25 degrees)
490  * where the 1st part is for direction 2. The 2nd and 3rd part are for direction 3,
491  * and the 4th part is for direction 4.
492  *
493  * Rather than re-inventing a solution, re-use the same checks as
494  * above with swapped xder and yder. */
495  if (yder * TAN11_25 < xder) {
496  yaw = 4;
497  } else if (yder * TAN33_75 < xder) {
498  yaw = 3;
499  } else {
500  yaw = 2;
501  }
502  }
503  if (swap_dx) yaw = 8 - yaw;
504  if (swap_dy) yaw = (16 - yaw) & 0xf;
505 
506  xpos_back &= 0xFFFFFF00; // Back and front positions to the north bottom corner of the voxel.
507  ypos_back &= 0xFFFFFF00;
508  zpos_back &= 0xFFFFFF00;
509  xpos_front &= 0xFFFFFF00;
510  ypos_front &= 0xFFFFFF00;
511  zpos_front &= 0xFFFFFF00;
512 
513  car.back.Set( xpos_back >> 8, ypos_back >> 8, zpos_back >> 8, xpos_middle - xpos_back, ypos_middle - ypos_back, zpos_middle - zpos_back, pitch, roll, yaw);
514  car.front.Set(xpos_front >> 8, ypos_front >> 8, zpos_front >> 8, xpos_middle - xpos_front, ypos_middle - ypos_front, zpos_middle - zpos_front, pitch, roll, yaw);
515  position += this->coaster->car_type->inter_car_length;
516  }
517 }

690 bool CoasterInstance::MakePositionedPiecesLooping(bool *modified)
691 {
692  if (modified != nullptr) *modified = false;
693 
694  /* First step, move all non-null track pieces to the start of the array. */
695  int count = 0;
696  for (int i = 0; i < this->capacity; i++) {
697  PositionedTrackPiece *ptp = this->pieces + i;
698  if (ptp->piece == nullptr) continue;
699  if (i == count) {
700  count++;
701  continue;
702  }
703  std::swap(this->pieces[count], *ptp);
704  if (modified != nullptr) *modified = true;
705  ptp->piece = nullptr;
706  count++;
707  }
708 
709  /* Second step, find a loop from start to end. */
710  if (count < 2) return false; // 0 or 1 positioned pieces won't ever make a loop.
711 
712  PositionedTrackPiece *ptp = this->pieces;
713  uint32 distance = 0;
714  if (ptp->distance_base != distance) {
715  if (modified != nullptr) *modified = true;
716  ptp->distance_base = distance;
717  }
718  distance += ptp->piece->piece_length;
719 
720  for (int i = 1; i < count; i++) {
721  int j = this->FindSuccessorPiece(ptp->GetEndX(), ptp->GetEndY(), ptp->GetEndZ(), ptp->piece->exit_connect, i, count);
722  if (j < 0) return false;
723  ptp++; // Now points to pieces[i].
724  if (i != j) {
725  std::swap(*ptp, this->pieces[j]); // Make piece 'j' the next positioned piece.
726  if (modified != nullptr) *modified = true;
727  }
728  if (ptp->distance_base != distance) {
729  if (modified != nullptr) *modified = true;
730  ptp->distance_base = distance;
731  }
732  distance += ptp->piece->piece_length;
733  }
734  this->coaster_length = distance;
735  return this->pieces[0].CanBeSuccessor(*ptp);
736 }
