114 void PathSearcher::AddOpen(int x, int y, int z, uint32 traveled, const WalkedPosition *prev_pos)
115 {
116  uint32 estimate = this->GetEstimate(x, y, z);
117  WalkedPosition wp(x, y, z, traveled, estimate, prev_pos);
118 
119  /* Find the position. */
120  PositionSet::iterator pos_iter = this->positions.find(wp);
121  if (pos_iter == positions.end()) { // New position.
122  pos_iter = this->positions.insert(wp).first;
123  this->open_points.emplace(traveled, estimate, &(*pos_iter));
124  return;
125  }
126  /* Existing position, update if needed. */
127  const WalkedPosition *pwp = &(*pos_iter);
128  if (pwp->traveled + pwp->estimate <= traveled + estimate) return;
129 
130  /* New one is better, update. */
131  pwp->traveled = traveled;
132  pwp->estimate = estimate; // The sum is changed, making any old open points invalid.
133  pwp->prev_pos = prev_pos;
134  this->open_points.emplace(traveled, estimate, pwp);
135 }

141 bool PathSearcher::Search()
142 {
143  this->dest_pos = nullptr;
144  while (!open_points.empty()) {
145  WalkedDistance wd = *open_points.begin();
146  open_points.erase(open_points.begin());
147 
148  if (wd.traveled != wd.pos->traveled || wd.estimate != wd.pos->estimate) continue; // Invalid open point.
149 
150  /* Reached the destination? */
151  const WalkedPosition *wp = wd.pos;
152  if (wp->x == this->dest_x && wp->y == this->dest_y && wp->z == this->dest_z) {
153  this->dest_pos = wp;
154  return true;
155  }
156 
157  /* Add new open points. */
158  const Voxel *v = _world.GetVoxel(wp->x, wp->y, wp->z);
159  if (v == nullptr) continue; // No voxel at the expected point, don't bother.
160 
161  uint8 exits = GetPathExits(v);
162  for (TileEdge edge = EDGE_BEGIN; edge < EDGE_COUNT; edge++) {
163  if ((exits & (0x11 << edge)) == 0) continue;
164 
165  /* There is an outgoing connection, is it also on the world? */
166  Point16 dxy = _tile_dxy[edge];
167  if (dxy.x < 0 && wp->x == 0) continue;
168  if (dxy.x > 0 && wp->x + 1 == _world.GetXSize()) continue;
169  if (dxy.y < 0 && wp->y == 0) continue;
170  if (dxy.y > 0 && wp->y + 1 == _world.GetYSize()) continue;
171 
172  int new_z = ((exits & (0x10 << edge)) != 0) ? wp->z + 1 : wp->z;
173  if (new_z < 0 || new_z >= WORLD_Z_SIZE) continue;
174 
175  /* Now check the other side, new_z is the voxel where the path should be at the bottom. */
176  const Voxel *v2 = _world.GetVoxel(wp->x + dxy.x, wp->y + dxy.y, new_z);
177  if (v2 == nullptr) continue;
178 
179  uint8 other_exits = GetPathExits(v2);
180  if ((other_exits & (1 << ((edge + 2) % 4))) == 0) { // No path here, try one voxel below
181  new_z--;
182  if (new_z < 0) continue;
183  v2 = _world.GetVoxel(wp->x + dxy.x, wp->y + dxy.y, new_z);
184  if (v2 == nullptr) continue;
185  other_exits = GetPathExits(v2);
186  if ((other_exits & (0x10 << ((edge + 2) % 4))) == 0) continue;
187  }
188  /* Add new open point to the path finder. */
189  this->AddOpen(wp->x + dxy.x, wp->y + dxy.y, new_z, wp->traveled + 1, wp);
190  }
191  }
192  return false;
193 }
