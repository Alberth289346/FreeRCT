210  void RestrictTo(typename PT::CoordType startx, typename PT::CoordType starty, SZ w, SZ h)
211  {

213 
214  typename PT::CoordType xend = this->base.x + this->width;
215  if (xend > static_cast<typename PT::CoordType>(startx + w)) xend = startx + w;
216  if (this->base.x < startx) this->base.x = startx;
217  this->width = (this->base.x < xend) ? xend - this->base.x : 0;
218 
219  typename PT::CoordType yend = this->base.y + this->height;
220  if (yend > static_cast<typename PT::CoordType>(starty + h)) yend = starty + h;
221  if (this->base.y < starty) this->base.y = starty;
222  this->height = (this->base.y < yend) ? yend - this->base.y : 0;
