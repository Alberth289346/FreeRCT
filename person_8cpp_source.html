62  length--;
63 
64  if (length != 13 * count) return false;

389 RideVisitDesire Guest::ComputeExitDesire(TileEdge current_edge, int x, int y, int z, TileEdge exit_edge, bool *seen_wanted_ride)
390 {
391  if (current_edge == exit_edge) return RVD_NO_VISIT; // Skip incoming edge (may get added later if no other options exist).
392 
393  bool travel = TravelQueuePath(&x, &y, &z, &exit_edge);
394  if (!travel) return RVD_NO_VISIT; // Path leads to nowhere.
395 
396  if (PathExistsAtBottomEdge(x, y, z, exit_edge)) return RVD_NO_RIDE; // Found a path.
397 
398  const RideInstance *ri = RideExistsAtBottom(x, y, z, exit_edge);
399  if (ri == nullptr || ri->state != RIS_OPEN) return RVD_NO_VISIT; // No ride, or a closed one.
400 
401  if (ri == this->wants_visit) { // Guest decided before that this shop/ride should be visited.
402  *seen_wanted_ride = true;
403  return RVD_MUST_VISIT;
404  }
405 
406  Point16 dxy = _tile_dxy[exit_edge];
407  if (!ri->CanBeVisited(x + dxy.x, y + dxy.y, z, exit_edge)) return RVD_NO_VISIT; // Ride cannot be entered here.
408 
409  RideVisitDesire rvd = this->WantToVisit(ri);
410  if ((rvd == RVD_MAY_VISIT || rvd == RVD_MUST_VISIT) && this->wants_visit == nullptr) {
411  /* Decided to want to visit one ride, and no wanted ride yet. */
412  // \todo Add a timeout so a guest gets bored waiting for the ride at some point.
413  this->wants_visit = ri;
414  *seen_wanted_ride = true;
415  return RVD_MUST_VISIT;
416  }
417  return rvd;
418 }

429 uint8 Guest::GetExitDirections(const Voxel *v, TileEdge start_edge, bool *seen_wanted_ride, bool *queue_mode)
430 {
431  assert(HasValidPath(v));
432 
433  /* If walking on a queue path, enable queue mode. */
434  // \todo Only walk in queue mode when going to a ride.
435  *queue_mode = _sprite_manager.GetPathStatus(GetPathType(v->GetInstanceData())) == PAS_QUEUE_PATH;
436  *seen_wanted_ride = false;
437 
438  uint8 bot_exits = 0xF; // All exit directions at the bottom by default.
439  uint8 shops = 0; // Number of exits with a shop with normal desire to go there.
440  uint8 top_exits = 0; // Exits at the top of the voxel.
441  uint8 must_shops = 0; // Shops with a high desire to visit.
442 
443  uint8 slope = GetImplodedPathSlope(v);
444  if (slope < PATH_FLAT_COUNT) { // At a flat path tile.
445  bot_exits &= _path_expand[slope] >> PATHBIT_NE; // Masks to exits only, as that's what 'bot_exits' contains here.
446  } else { // At a path ramp.
447  switch (slope) {
448  case PATH_RAMP_NE: bot_exits = 1 << EDGE_NE; top_exits = 1 << EDGE_SW; break;
449  case PATH_RAMP_NW: bot_exits = 1 << EDGE_NW; top_exits = 1 << EDGE_SE; break;
450  case PATH_RAMP_SE: bot_exits = 1 << EDGE_SE; top_exits = 1 << EDGE_NW; break;
451  case PATH_RAMP_SW: bot_exits = 1 << EDGE_SW; top_exits = 1 << EDGE_NE; break;
452  default: NOT_REACHED();
453  }
454  }
455 
456  /* Being at a path tile, make extra sure we don't leave the path. */
457  for (TileEdge exit_edge = EDGE_BEGIN; exit_edge != EDGE_COUNT; exit_edge++) {
458  int z; // Decide z position of the exit.
459  if (GB(bot_exits, exit_edge, 1) != 0) {
460  z = this->z_vox;
461  } else if (GB(top_exits, exit_edge, 1) != 0) {
462  z = this->z_vox + 1;
463  } else {
464  continue;
465  }
466 
467  RideVisitDesire rvd = ComputeExitDesire(start_edge, this->x_vox, this->y_vox, z, exit_edge, seen_wanted_ride);
468  switch (rvd) {
469  case RVD_NO_RIDE:
470  break; // A path is one of the options.
471 
472  case RVD_NO_VISIT: // No desire to visit this exit; clear it.
473  SB(bot_exits, exit_edge, 1, 0);
474  SB(top_exits, exit_edge, 1, 0);
475  break;
476 
477  case RVD_MAY_VISIT: // It's one of the options (since the person is not coming from it).
478  SB(shops, exit_edge, 1, 1);
479  break;
480 
481  case RVD_MUST_VISIT: // It is very desirable to visit this shop (since the person is not coming from it).
482  SB(must_shops, exit_edge, 1, 1);
483  break;
484 
485  default: NOT_REACHED();
486  }
487  }
488  bot_exits |= top_exits; // Difference between bottom-exit and top-exit is not relevant any more.
489  if (must_shops != 0) { // If there are shops that must be visited, ignore everything else.
490  bot_exits &= must_shops;
491  shops = must_shops;
492  }
493  return (shops << 4) | bot_exits;
494 }

571 void Guest::DecideMoveDirection()
572 {
573  const VoxelStack *vs = _world.GetStack(this->x_vox, this->y_vox);
574  const Voxel *v = vs->Get(this->z_vox);
575  TileEdge start_edge = this->GetCurrentEdge(); // Edge the person is currently.
576 
577  if (this->activity == GA_ENTER_PARK && vs->owner == OWN_PARK) {
578  // \todo Pay the park fee, go home if insufficient monies.
579  this->activity = GA_WANDER;
580  // Add some happiness?? (Somewhat useless as every guest enters the park. On the other hand, a nice point to configure difficulty level perhaps?)
581  }
582 
583  /* Find feasible exits and shops. */
584  uint8 exits, shops;
585  if (HasValidPath(v)) {
586  bool seen_wanted_ride, queue_mode;
587  exits = GetExitDirections(v, start_edge, &seen_wanted_ride, &queue_mode);
588  shops = exits >> 4;
589  exits &= 0xF;
590 
591  this->queue_mode = queue_mode;
592  if (!seen_wanted_ride) this->wants_visit = nullptr; // Wanted ride has gone missing, stop looking for it.
593  } else { // Not at a path -> lost.
594  exits = 0xF;
595  shops = 0;
596  this->queue_mode = false;
597  this->wants_visit = nullptr;
598  }
599 
600  if (this->activity == GA_WANDER) { // Prevent wandering guests from walking out the park.
601  for (TileEdge exit_edge = EDGE_BEGIN; exit_edge != EDGE_COUNT; exit_edge++) {
602  if (GB(exits, exit_edge, 1) == 0) continue;
603  if (_world.GetTileOwner(this->x_vox + _tile_dxy[exit_edge].x, this->y_vox + _tile_dxy[exit_edge].y) != OWN_PARK) {
604  SB(exits, exit_edge, 1, 0);
605  SB(shops, exit_edge, 1, 0);
606  }
607  }
608  }
609 
610  /* Decide which direction to go. */

698 AnimateResult Person::OnAnimate(int delay)
699 {
700  this->frame_time -= delay;
701  if (this->frame_time > 0) return OAR_OK;
702 
703  this->MarkDirty(); // Marks the entire voxel dirty, which should be big enough even after moving.
704 
705  if (this->frames == nullptr || this->frame_count == 0) return OAR_REMOVE;
706 
707  int16 x_limit = -1;
708  switch (GB(this->walk->limit_type, WLM_X_START, WLM_LIMIT_LENGTH)) {
709  case WLM_MINIMAL: x_limit = 0; break;
710  case WLM_LOW: x_limit = 128 - this->offset; break;
711  case WLM_CENTER: x_limit = 128; break;
712  case WLM_HIGH: x_limit = 128 + this->offset; break;
713  case WLM_MAXIMAL: x_limit = 255; break;
714  }
715 
716  int16 y_limit = -1;
717  switch (GB(this->walk->limit_type, WLM_Y_START, WLM_LIMIT_LENGTH)) {
718  case WLM_MINIMAL: y_limit = 0; break;
719  case WLM_LOW: y_limit = 128 - this->offset; break;
720  case WLM_CENTER: y_limit = 128; break;
721  case WLM_HIGH: y_limit = 128 + this->offset; break;
722  case WLM_MAXIMAL: y_limit = 255; break;
723  }
724 
725  uint16 index = this->frame_index;
726  const AnimationFrame *frame = &this->frames[index];
727  this->x_pos += frame->dx;
728  this->y_pos += frame->dy;
729 
730  bool reached = false; // Set to true when we are beyond the limit!
731  if ((this->walk->limit_type & (1 << WLM_END_LIMIT)) == WLM_X_COND) {
732  if (frame->dx > 0) reached |= this->x_pos > x_limit;
733  if (frame->dx < 0) reached |= this->x_pos < x_limit;
734 
735  if (y_limit >= 0) this->y_pos += sign(y_limit - this->y_pos); // Also slowly move the other axis in the right direction.
736  } else {
737  if (frame->dy > 0) reached |= this->y_pos > y_limit;
738  if (frame->dy < 0) reached |= this->y_pos < y_limit;
739 
740  if (x_limit >= 0) this->x_pos += sign(x_limit - this->x_pos); // Also slowly move the other axis in the right direction.
741  }
742 
743  if (!reached) {
744  /* Not reached the end, do the next frame. */
745  index++;
746  if (this->frame_count <= index) index = 0;
747  this->frame_index = index;
748  this->frame_time = this->frames[index].duration;
749 
750  this->z_pos = GetZHeight(this->x_vox, this->y_vox, this->z_vox, this->x_pos, this->y_pos);
751  return OAR_OK;
752  }
753 
754  /* Reached the goal, start the next walk. */
755  if (this->walk[1].anim_type != ANIM_INVALID) {
756  this->StartAnimation(this->walk + 1);
757  return OAR_OK;
758  }
759 
760  /* Not only the end of this walk, but the end of the entire walk at the tile. */
761  int dx = 0;
762  int dy = 0;
763  int dz = 0;
764  TileEdge exit_edge = INVALID_EDGE;
765 
766  this->RemoveSelf(_world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false));
767  if (this->x_pos < 0) {
768  dx--;
769  this->x_vox--;
770  this->x_pos += 256;
771  exit_edge = EDGE_NE;
772  } else if (this->x_pos > 255) {
773  dx++;
774  this->x_vox++;
775  this->x_pos -= 256;
776  exit_edge = EDGE_SW;
777  }
778  if (this->y_pos < 0) {
779  dy--;
780  this->y_vox--;
781  this->y_pos += 256;
782  exit_edge = EDGE_NW;
783  } else if (this->y_pos > 255) {
784  dy++;
785  this->y_vox++;
786  this->y_pos -= 256;
787  exit_edge = EDGE_SE;
788  }
789  assert(this->x_pos >= 0 && this->x_pos < 256);
790  assert(this->y_pos >= 0 && this->y_pos < 256);
791 
792  /* If the guest ended up off-world, quit. */
793  if (this->x_vox < 0 || this->x_vox >= _world.GetXSize() * 256 ||
794  this->y_vox < 0 || this->y_vox >= _world.GetYSize() * 256) {
795  return OAR_DEACTIVATE;
796  }
797 
798  /* Handle raising of z position. */
799  if (this->z_pos > 128) {
800  dz++;
801  this->z_vox++;
802  this->z_pos = 0;
803  }
804  /* At bottom of the voxel. */
805  Voxel *v = _world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false);
806  if (v != nullptr) {
807  SmallRideInstance instance = v->GetInstance();
808  if (instance >= SRI_FULL_RIDES) {
809  assert(exit_edge != INVALID_EDGE);
810  RideInstance *ri = _rides_manager.GetRideInstance(instance);
811  if (ri->CanBeVisited(this->x_vox, this->y_vox, this->z_vox, exit_edge)) {
812  Guest *guest = dynamic_cast<Guest *>(this);
813  if (guest != nullptr) guest->VisitShop(ri);
814  }
815  /* Ride is closed, or not a guest, fall-through to reversing movement. */
816 
817  } else if (HasValidPath(v)) {
818  this->AddSelf(v);
819  this->DecideMoveDirection();
820  return OAR_OK;
821  }
822 
823  /* Maybe a path below this voxel? */
824  if (this->z_vox > 0) {
825  dz--;
826  this->z_vox--;
827  this->z_pos = 255;
828  Voxel *w = _world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false);
829  if (w != nullptr && HasValidPath(w)) {
830  this->AddSelf(w);
831  this->DecideMoveDirection();
832  return OAR_OK;
833  }
834  }
835 
836  /* Restore the person at the previous tile (ie reverse movement). */
837  if (dx != 0) { this->x_vox -= dx; this->x_pos = (dx > 0) ? 255 : 0; }
838  if (dy != 0) { this->y_vox -= dy; this->y_pos = (dy > 0) ? 255 : 0; }
839  if (dz != 0) { this->z_vox -= dz; this->z_pos = (dz > 0) ? 255 : 0; }
840 
841  this->AddSelf(_world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false));
842  this->DecideMoveDirection();
843  return OAR_OK;
844  }
845  /* No voxel here, try one level below. */
846  if (this->z_vox > 0) {
847  dz--;
848  this->z_vox--;
849  this->z_pos = 255;
850  v = _world.GetCreateVoxel(this->x_vox, this->y_vox, this->z_vox, false);
851  }
852  if (v != nullptr && HasValidPath(v)) {
853  this->AddSelf(v);
854  this->DecideMoveDirection();
855  return OAR_OK;
856  }
857  return OAR_DEACTIVATE; // We are truly lost now.
858 }
